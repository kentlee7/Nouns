-- Ratio of bare sg and pl

SELECT
    lemma,
    SUM(CASE WHEN definiteness = 'barePl' THEN 1 ELSE 0 END) AS bare_pl_tokens,
    SUM(CASE WHEN definiteness = 'bareSg' THEN 1 ELSE 0 END) AS bare_sg_tokens,
    ROUND(
        SUM(CASE WHEN definiteness = 'barePl' THEN 1 ELSE 0 END) * 1.0
        /
        NULLIF(SUM(CASE WHEN definiteness = 'bareSg' THEN 1 ELSE 0 END), 0),
        4
    ) AS pl_sg_ratio
FROM nouns
WHERE definiteness IN ('bareSg', 'barePl')
GROUP BY lemma
HAVING
    SUM(CASE WHEN definiteness = 'barePl' THEN 1 ELSE 0 END) > 0
    AND
    SUM(CASE WHEN definiteness = 'bareSg' THEN 1 ELSE 0 END) > 0
ORDER BY pl_sg_ratio DESC;


-- Proportion of bare sg and pl  +
-- saved as prop_definiteness_barePl_bareSg.csv

SELECT
    lemma,
    COUNT(*) AS total_bare_tokens,
    SUM(CASE WHEN definiteness = 'barePl' THEN 1 ELSE 0 END) AS bare_pl_tokens,
    ROUND(
        SUM(CASE WHEN definiteness = 'barePl' THEN 1 ELSE 0 END) * 1.0
        / COUNT(*),
        4
    ) AS bare_pl_proportion
FROM nouns
WHERE definiteness IN ('barePl', 'bareSg')
GROUP BY lemma
HAVING COUNT(*) >= 5
ORDER BY bare_pl_proportion DESC;



-- Proportion of nouns classified by script as count / mass +
-- saved as prop_massct_countability_ct_mass.csv

SELECT
    lemma,
    COUNT(*) AS massct_tokens,
    SUM(CASE WHEN countability = 'count' THEN 1 ELSE 0 END) AS ct_tokens,
    ROUND(
        SUM(CASE WHEN countability = 'count' THEN 1 ELSE 0 END) * 1.0
        / COUNT(*),
        4
    ) AS ct_proportion
FROM nouns
WHERE countability IN ('count', 'mass')
GROUP BY lemma
HAVING COUNT(*) >= 5
ORDER BY ct_proportion DESC;


-- Flexibility index based on bare singular vs. bare plural
-- flex_index = barePl / (barePl + bareSg)
-- 1.0 = strongly count N; 0.5 = strongly flex N; 0.0 = strongly masss N
-- saved as flex_bareNP_index.csv

SELECT
    lemma,
    SUM(CASE WHEN definiteness = 'barePl' THEN 1 ELSE 0 END) AS bare_pl,
    SUM(CASE WHEN definiteness = 'bareSg' THEN 1 ELSE 0 END) AS bare_sg,
    ROUND(
        SUM(CASE WHEN definiteness = 'barePl' THEN 1 ELSE 0 END) * 1.0
        /
        NULLIF(
            SUM(CASE WHEN definiteness IN ('barePl', 'bareSg') THEN 1 ELSE 0 END),
            0
        ),
        4
    ) AS flex_bareN_index
FROM nouns
WHERE definiteness IN ('barePl', 'bareSg')
GROUP BY lemma
HAVING COUNT(*) >= 5
ORDER BY flex_bareN_index DESC;



-- Flexibility index based on countability annotation
-- flex_index = count / (count + mass)
-- 1.0 = strongly count N; 0.5 = strongly flex N; 0.0 = strongly masss N
-- saved as flex_assignedNP_index.csv

SELECT
    lemma,
    SUM(CASE WHEN countability = 'count' THEN 1 ELSE 0 END) AS ct_tokens,
    SUM(CASE WHEN countability = 'mass' THEN 1 ELSE 0 END) AS mass_tokens,
    ROUND(
        SUM(CASE WHEN countability = 'count' THEN 1 ELSE 0 END) * 1.0
        /
        NULLIF(
            SUM(CASE WHEN countability IN ('count', 'mass') THEN 1 ELSE 0 END),
            0
        ),
        4
    ) AS flex_assgd_index
FROM nouns
WHERE countability IN ('count', 'mass')
GROUP BY lemma
HAVING COUNT(*) >= 5
ORDER BY flex_assgd_index DESC;


-- compound nouns
-- saved as compound_NOUN_counts.csv

SELECT
    lemma,
    COUNT(*) AS total_tokens,

    -- compound vs non-compound
    SUM(CASE WHEN number LIKE 'cpd_%' THEN 1 ELSE 0 END) AS compound_tokens,
    SUM(CASE WHEN number NOT LIKE 'cpd_%' THEN 1 ELSE 0 END) AS non_compound_tokens,

    -- compound breakdown
    SUM(CASE WHEN number = 'cpd_nonhead' THEN 1 ELSE 0 END) AS nonhead_tokens,
    SUM(CASE WHEN number IN ('cpd_head_sg','cpd_head_pl','cpd_proper')
        THEN 1 ELSE 0 END) AS head_tokens,

    -- ratios
    ROUND(
        SUM(CASE WHEN number LIKE 'cpd_%' THEN 1 ELSE 0 END) * 1.0
        / NULLIF(COUNT(*),0),
        4
    ) AS compound_ratio,

    ROUND(
        SUM(CASE WHEN number IN ('cpd_head_sg','cpd_head_pl','cpd_proper')
            THEN 1 ELSE 0 END) * 1.0
        /
        NULLIF(
            SUM(CASE WHEN number LIKE 'cpd_%' THEN 1 ELSE 0 END),
            0
        ),
        4
    ) AS head_ratio_within_compounds

FROM nouns
WHERE number NOT LIKE 'proper%'
GROUP BY lemma
HAVING COUNT(*) >= 10
ORDER BY compound_ratio DESC;


-- ixnay
-- Basic Compound Frequency / Productivity Index (CPI)
-- CPI= total noun tokens / compound tokens
​-- Interpretation:
-- 0.0 = never compounded; 1.0 = always compounded; 0.5 = half the time in compounds
-- saved as compound_freq_index.csv
-- ixnay

SELECT
    lemma,
    COUNT(*) AS total_tokens,

    SUM(CASE WHEN number LIKE 'cpd_%'
        THEN 1 ELSE 0 END) AS compound_tokens,

    ROUND(
        SUM(CASE WHEN number LIKE 'cpd_%' THEN 1 ELSE 0 END) * 1.0
        / NULLIF(COUNT(*),0),
        4
    ) AS compound_productivity_index

FROM nouns
GROUP BY lemma
HAVING COUNT(*) >= 20
ORDER BY compound_productivity_index DESC;


-- ixnay
-- Head Productivity vs Modifier Productivity
-- Modifier Index = nonhead tokens / total tokens
-- head index = head tokens / total tokent
-- saved as cpd_modifier_head_index.csv
-- ixnay

SELECT
    lemma,
    COUNT(*) AS total_tokens,

    SUM(CASE WHEN number = 'cpd_nonhead'
        THEN 1 ELSE 0 END) AS modifier_tokens,

    SUM(CASE WHEN number IN ('cpd_head_sg','cpd_head_pl','cpd_proper')
        THEN 1 ELSE 0 END) AS head_tokens,

    ROUND(
        SUM(CASE WHEN number = 'cpd_nonhead'
            THEN 1 ELSE 0 END) * 1.0
        / NULLIF(COUNT(*),0),
        4
    ) AS modifier_index,

    ROUND(
        SUM(CASE WHEN number IN ('cpd_head_sg','cpd_head_pl','cpd_proper')
            THEN 1 ELSE 0 END) * 1.0
        / NULLIF(COUNT(*),0),
        4
    ) AS head_index

FROM nouns
GROUP BY lemma
HAVING COUNT(*) >= 20
ORDER BY modifier_index DESC;

-- Left join noun data, starting with lexical concreteness ratings, for correlations

COMMIT;

SELECT c.lemma, c.bigram, c.conc_mean, c.conc_sd, c.lexfreq,
   bc.bare_bpl_tk, bc.bare_bsg_tk, bc.total_tk_bare, bc.pl_sg_prop,
   d.massct_tk, d.ct_tk, d.mass_tk, d.ct_proportion,
   fb.flex_bareN_index,
   fmc.flex_assgd_index
FROM concreteness c
LEFT JOIN bare_ct bc
ON c.lemma = bc.lemma
LEFT JOIN ct_mass_assigned d
ON c.lemma = d.lemma
LEFT JOIN flex_bare fb
ON c.lemma = fb.lemma
LEFT JOIN flex_massct fmc
ON c.lemma = fmc.lemma
;


-- Left join compound noun data, starting with lexical concreteness ratings, for correlations

SELECT c.lemma, c.bigram, c.conc_mean, c.conc_sd, c.lexfreq,
   b.bare_bpl_tk, b.bare_bsg_tk, b.total_tk_bare, b.pl_sg_prop,
   d.massct_tk, d.ct_tk, d.ct_proportion,
   cpd.cpd_tk, cpd.ncpd_tk, cpd.cpd_freq_index
FROM concreteness c
LEFT JOIN bare_ct b
ON c.lemma = b.lemma
LEFT JOIN ct_mass_assigned d
ON c.lemma = d.lemma
LEFT JOIN cpdfreq cpd
ON c.lemma = cpd.lemma
;
