
library(tidyverse) 
library(psych)
library(ggplot2)

# Import first file - general noun info including countability indices 

nouns <- read.csv("/home/kent/R/nouns/output/nouns_main_dataset_R_stats.csv", na.strings=".")
# Check the variable formats 
str(nouns)
colnames(nouns)

# Isolate the desired variables 

vars <- nouns[, c("conc_mean",
                 "conc_sd",
                 "lexfreq",
                 "ct_proportion",
                 "pl_sg_prop")]
# using psych
round(corr.test(vars, use="pairwise")$p, 4)

corr.test(vars, use="pairwise") 


# line plot - prob density plot for categorically assigned count noun index
ggplot(nouns, aes(x = ct_proportion)) +
  geom_density(
    fill = "steelblue",
    alpha = 0.6
  ) +
  scale_x_continuous(
    breaks = seq(0, 1, 0.1),
    limits = c(0, 1)
  ) +
  labs(
    x = "Countability Index (general)",
    y = "# Noun Lemmas x 1000",
    title = "Probability Density for General Noun Countability Index" 
  ) +
  theme_minimal()


# line plot - prob density plot for bare pl count noun index
